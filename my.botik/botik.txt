import telebot
from telebot import types

TOKEN = "BOT_TOKEN"
bot = telebot.TeleBot(TOKEN)

@bot.message_handler(commands=['start'])
def start(message):
    markup = types.ReplyKeyboardMarkup(resize_keyboard=True)
    btn = types.KeyboardButton("🌿 Показать реквизиты")
    markup.add(btn)

    bot.send_message(message.chat.id, "Выбери:", reply_markup=markup)

@bot.message_handler(func=lambda message: message.text == "🌿 Показать реквизиты")
def req(message):
    markup = types.ReplyKeyboardMarkup(resize_keyboard=True)
    btn = types.KeyboardButton("⬅️ Назад")
    markup.add(btn)

    bot.send_message(
        message.chat.id,
        "Карта Ю.money: 5599 0021 3673 1108",
        reply_markup=markup
    )

@bot.message_handler(func=lambda message: message.text == "⬅️ Назад")
def back(message):
    start(message)

bot.polling(none_stop=True)
